const SUPABASE_KEY = 'sb_publishable_r7Bgf2u5ZK30-DQVHPSHHQ_xgnsXY2x'; const supabase = createClient(SUPABASE_URL, SUPABASE_KEY); // --- Глобални променливи --- let transactions = []; let categoriesMemory = {}; // автоматично запомняне на категории/клиенти/методи let currentProfile = 'Личен'; // default let currentMonth = new Date().getMonth() + 1; // текущ месец // --- UI --- const incomeColumn = document.getElementById('income-column'); const expenseColumn = document.getElementById('expense-column'); const incomeChartCanvas = document.getElementById('income-chart'); const expenseChartCanvas = document.getElementById('expense-chart'); const profileSelect = document.getElementById('profile-select'); const monthSelect = document.getElementById('month-select'); // --- Функции за добавяне на транзакции --- async function addTransaction(type, category, method, amount, object) { if (!amount || isNaN(amount)) return alert('Моля въведете сума'); const { data, error } = await supabase .from('transactions') .insert([{ type, category, method, amount, object, profile: currentProfile, created_at: new Date() }]); if (error) return console.error(error); transactions.push(data[0]); updateUI(); memorizeCategory(type, category, method); } // --- Функция за запомняне на категории --- function memorizeCategory(type, category, method) { const key = `${type}_${category}_${method}`; categoriesMemory[key] = (categoriesMemory[key] || 0) + 1; } // --- Функция за изтриване --- async function deleteTransaction(id) { const { error } = await supabase.from('transactions').delete().eq('id', id); if (error) return console.error(error); transactions = transactions.filter(t => t.id !== id); updateUI(); } // --- Функция за визуализация --- function updateUI() { incomeColumn.innerHTML = ''; expenseColumn.innerHTML = ''; const incomeData = {}; const expenseData = {}; transactions.forEach(t => { if (t.type === 'Доход' || t.type === 'Бонус') { const key = `${t.category} (${t.method})`; incomeData[key] = (incomeData[key] || 0) + parseFloat(t.amount); } else { const key = `${t.category} (${t.method})`; expenseData[key] = (expenseData[key] || 0) + parseFloat(t.amount); } }); // render редове Object.entries(incomeData).forEach(([k, v]) => { const row = document.createElement('div'); row.className = 'row'; row.innerHTML = `${k}: ${v.toFixed(2)} € `; incomeColumn.appendChild(row); }); Object.entries(expenseData).forEach(([k, v]) => { const row = document.createElement('div'); row.className = 'row'; row.innerHTML = `${k}: ${v.toFixed(2)} € `; expenseColumn.appendChild(row); }); renderCharts(incomeData, expenseData); } // --- Функция за графики --- function renderCharts(incomeData, expenseData) { new Chart(incomeChartCanvas, { type: 'bar', data: { labels: Object.keys(incomeData), datasets: [{ label: 'Доход + Бонус', data: Object.values(incomeData), backgroundColor: '#4CAF50' }] }, options: { responsive: true, maintainAspectRatio: false } }); new Chart(expenseChartCanvas, { type: 'bar', data: { labels: Object.keys(expenseData), datasets: [{ label: 'Разходи', data: Object.values(expenseData), backgroundColor: '#F44336' }] }, options: { responsive: true, maintainAspectRatio: false } }); } // --- Зареждане на транзакции от Supabase --- async function loadTransactions() { const { data, error } = await supabase .from('transactions') .select('*') .eq('profile', currentProfile) .gte('extract(month from created_at)', currentMonth); if (error) return console.error(error); transactions = data; updateUI(); } // --- Initial --- loadTransactions();
